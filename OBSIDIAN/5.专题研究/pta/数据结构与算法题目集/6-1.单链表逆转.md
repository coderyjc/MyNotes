List Reverse( List L ){
    PtrToNode p = L;
    if(L == NULL || L->Next == NULL) return p;
    PtrToNode q = p->Next;
    p->Next = NULL;
    PtrToNode t = q->Next;
    do{
        q->Next = p;
        p = q;
        q = t;
        if(t != NULL) t = t->Next;
    }while(q);
    PtrToNode rst = p;
    return rst;
}
