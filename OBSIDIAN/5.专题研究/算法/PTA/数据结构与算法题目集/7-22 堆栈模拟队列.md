关键点：

往栈的容量小的那个容器当中存入数据，  

当其达到容量时，就将其中的数据转到另一个容器当中，

```c
#include<bits/stdc++.h>

using namespace std;

int main(){
//  初始化
    int c1, c2;
    stack<int> s1;
    stack<int> s2;
    cin >> c1 >> c2;
    if(c1 > c2) swap(c1, c2);

//  指令
    char cmd = 0;
    int temp;
    while(cmd != 'T'){
        cin >> cmd;
        if('A' == cmd){
            cin >> temp;
            if(s1.size() < c1) s1.push(temp);
            else if (s1.size() == c1 && s2.size() == 0){
                while(!s1.empty()){
                    s2.push(s1.top());
                    s1.pop();
                }
                s1.push(temp);
            } else {
                cout << "ERROR:Full" << endl;
            }
        }
        else if('D' == cmd){
            if(s2.size() > 0){
                cout << s2.top() << endl;
                s2.pop();
            } else if(!s1.empty()) {
                while(!s1.empty()){
                    s2.push(s1.top());
                    s1.pop();
                }
                cout << s2.top() << endl;
                s2.pop();
            } else cout << "ERROR:Empty" << endl;
        }
    }

    return 0;
}
```
