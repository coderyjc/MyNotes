```c
#include<bits/stdc++.h>

using namespace std;

int main() {

    int n, s, t;
    cin >> n >> s;
    string name;
    map<int, vector<string>> m;
    int subject;
    
    for (int i = 0; i < n; ++i) {
        cin >> name >> t;
        for (int j = 0; j < t; ++j) {
            scanf("%d", &subject);
            m[subject].push_back(name);
        }
    }

    for (int i = 1; i <= s; ++i) {
        printf("%d %zu\n", i, m[i].size());
        sort(m[i].begin(), m[i].end());
        for (int j = 0; j < m[i].size(); ++j) {
            printf("%s\n", m[i][j].c_str());
        }
    }
    return 0;
}
```
