```c
#include <cstdio>  
#include <cstring>  
  
#define MAXSIZE_PLUS 1001  
#define MAXSIZE_MULT 2001  
  
void printResult(int result[], int n){  
    // 输出标志，i为指数，rst_plus[i]为系数  
    int flag = 0;  
    for (int i = n - 1; i >= 0; i--) {  
        if(result[i] != 0){  
            if(!flag){  
                printf("%d %d", result[i], i);  
                flag = 1;  
            } else {  
                printf(" %d %d", result[i], i);  
            }  
        }  
    }  
    if(!flag) printf("0 0");  
}  

int main() {  
    // 声明+加法  
    int a[MAXSIZE_PLUS], b[MAXSIZE_PLUS], rst_plus[MAXSIZE_PLUS], rst_mul[MAXSIZE_MULT];  
    memset(a, 0, sizeof(a));  
    memset(b, 0, sizeof(b));  
    memset(rst_plus, 0, sizeof(rst_plus));  
    memset(rst_mul, 0, sizeof(rst_mul));  
  
    // 输入，顺便进行加法  
    int n1, first, second;  
    scanf("%d", &n1);  
    for (int i = 0; i < n1; ++i) {  
        scanf("%d%d", &first, &second);  
        a[second] = first;  
        rst_plus[second] = first;  
    }  
    int n2;  
    scanf("%d", &n2);  
    for (int i = 0; i < n2; ++i) {  
        scanf("%d%d", &first, &second);  
        b[second] = first;  
        rst_plus[second] += first;  
    }  
    // 进行乘法计算，把a乘到b  
    for (int i = 0; i < MAXSIZE_PLUS; ++i) {  
        for (int j = 0; j < MAXSIZE_PLUS; ++j) {  
            rst_mul[i + j] += a[i] * b[j];  
        }  
    }  
    // 输出结果  
    printResult(rst_mul, MAXSIZE_MULT);  
    printf("\n");  
    printResult(rst_plus, MAXSIZE_PLUS);  
  
    return 0;  
}
```

直接使用两个数组存储结果即可。

加法是对应位置直接相加

乘法是一个多项式的每一项分别乘另一个的所有项，再把这些撑出来的结果多项式想加。
