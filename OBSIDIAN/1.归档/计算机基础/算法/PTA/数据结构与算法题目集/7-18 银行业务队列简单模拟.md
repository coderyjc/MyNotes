#include "iostream"
#include<cstdio>
#include<queue>

using namespace std;

// 8 2 1 3 9 4 11 13 15

int main(){

    int n, temp;
    cin >> n;
    queue<int> a, b;
    //  输入
    for (int i = 0; i < n; ++i) {
        cin >> temp;
        if(temp % 2) a.push (temp);
        else b.push(temp);
    }
    // 处理
    int flag = 0;
    while(!a.empty() || !b.empty()){
        if(!a.empty()){
            if (flag) printf(" ");
            cout << a.front();
            flag = 1;
            a.pop();
            if(!a.empty()){
                if (flag) printf(" ");
                cout << a.front();
                flag = 1;
                a.pop();
            }
        }
        if(!b.empty()){
            if (flag) printf(" ");
            cout << b.front();
            flag = 1;
            b.pop();
        }
    }

    return 0;
}
