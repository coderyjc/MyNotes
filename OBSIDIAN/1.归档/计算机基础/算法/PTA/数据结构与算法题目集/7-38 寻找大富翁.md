要考虑N < M的情况

```c
#include <iostream>
#include <cstdio>
#include <deque>
#include "algorithm"

using namespace std;

class compare{
public:
    bool operator()(int v1, int v2){
        return v1 < v2;
    }
};

int main() {

    int n, m, temp;
    deque<int> d;
    cin >> n >> m;
    int min;
    for (int i = 0; i < n; ++i) {
        cin >> temp;
        if(d.size() < 10) d.push_back(temp);
        else {
            sort(d.begin(), d.end(), compare());
            min = d[0];
            if(min < temp) {
                d.pop_front();
                d.push_back(temp);
            }
        }
    }
    sort(d.begin(), d.end(), compare());
    int flag = 0;
    if(n < m) m = n;
    for (int i = 0; i < m; ++i) {
        if(flag) cout << ' ';
        cout << d[d.size() - i - 1];
        flag = 1;
    }

    return 0;
}
```
