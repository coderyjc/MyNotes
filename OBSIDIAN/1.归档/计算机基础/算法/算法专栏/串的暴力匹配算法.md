## 简单的串匹配的算法

```c
#include <stdio.h>  
#include <string.h>  
  
#define MAXLEN 255 // 预定义最大长度为255  
typedef struct {  
    char ch[MAXLEN];  
    int length;  
}SString;  
  
//  假设i和j的下标都从0开始。  
int index(SString S, SString T){  
    int i = 0, j = 0;  
    while(i < S.length && j < T.length){  
        if(S.ch[i] == T.ch[j]){  
            ++i;  
            ++j;  
        } else {  
            i = i - j + 1;  
            j = 0;  
        }    }    if (j == T.length) return i - T.length;  
    else return 0;  
}  

int main() {  
  
    SString str1, str2;  
    strcpy(str1.ch, "ababcabcacbab");  
    str1.length = strlen(str1.ch);  
    strcpy(str2.ch, "abcac");  
    str2.length = strlen(str2.ch);  
  
    printf("%d", index(str1, str2)); // 输出5  
  
    return 0;  
}
```

从头开始查找，成功则向后移动，失败则：`i = i - j + 1;  j = 0; `
