void quickSort(int num[], int start, int end){
  int i = start, j = end;
  if(i >= j) return;
  int val = num[i];
  while (i < j) {
    // back to front, find small
    while(i < j && num[j] > val) j--;
    // find it
    if(i < j) num[i++] = num[j];
    // front to back, find bigger
    while(i < j && num[i] < val) i++;
    // find it
    if(i < j) num[j--] = num[i];
  }

  // i == j
  num[i] = val;
  quickSort(num, start, i - 1);
  quickSort(num, j + 1, end);
}
